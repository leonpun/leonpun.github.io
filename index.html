‘’’
/*
public class ListNode {
int val;
ListNode next = null;

ListNode(int val) {
    this.val = val;
}

}
*/
public class Solution {

public ListNode EntryNodeOfLoop(ListNode pHead)
{
    if(pHead == null) {
        return null;
    }
    ListNode fast = pHead;
    ListNode slow = pHead; 
    while(fast != null && fast.next != null) {
        fast = fast.next.next;
        slow = slow.next;

        if(fast == slow) {
            slow = pHead;

            while(slow != fast) {
                slow = slow.next;
                fast = fast.next;
            }
            return slow;
        }
    }
    return null;
}

}
